#!/usr/bin/env bash
set -Eeuo pipefail

# Всё в одном:
# - устанавливает Kitty, Firefox, Fuzzel и wev;
# - включает Hyprland через UWSM;
# - включает автоматический вход пользователя w3r через greetd;
# - делает Command похожим на Command в macOS:
#     Command+C/V/X/A/Z/S/F/W/T и т. д. -> обычные Ctrl-сочетания;
#     Command+Q                     -> закрыть активное приложение;
#     Command+Enter                 -> открыть Kitty;
#     Command+Space                 -> открыть меню приложений;
#     Control+Space                 -> переключить RU/EN;
# - настраивает копирование и вставку в Kitty.
#
# Запуск:
#   chmod +x setup-mac-hyprland.sh
#   sudo ./setup-mac-hyprland.sh
#
# При проблемах с keyd аварийная комбинация:
#   Backspace + Escape + Enter

TARGET_USER="${TARGET_USER:-${SUDO_USER:-w3r}}"
FLAKE_HOST="${FLAKE_HOST:-mbp}"

NIXOS_DIR="/etc/nixos"
SYSTEM_MODULE="$NIXOS_DIR/hypr-autostart.nix"
CONFIGURATION="$NIXOS_DIR/configuration.nix"

HYPR_DIR="$NIXOS_DIR/config/sessions/hyprland"
HYPR_MAIN="$HYPR_DIR/hyprland.lua"
HYPR_LUA_DIR="$HYPR_DIR/lua"
MAC_BINDS="$HYPR_LUA_DIR/mac-binds.lua"

if [[ "$EUID" -ne 0 ]]; then
    echo "Запусти скрипт через sudo."
    exit 1
fi

if ! id "$TARGET_USER" >/dev/null 2>&1; then
    echo "Пользователь $TARGET_USER не найден."
    exit 1
fi

USER_HOME="$(getent passwd "$TARGET_USER" | cut -d: -f6)"
USER_GROUP="$(id -gn "$TARGET_USER")"

if [[ ! -f "$NIXOS_DIR/flake.nix" ]]; then
    echo "Не найден $NIXOS_DIR/flake.nix"
    exit 1
fi

if [[ ! -f "$CONFIGURATION" ]]; then
    echo "Не найден $CONFIGURATION"
    exit 1
fi

STAMP="$(date +%Y%m%d-%H%M%S)"
BACKUP="/root/w3r-mac-binds-backup-$STAMP"

mkdir -p "$BACKUP"

cp -a "$CONFIGURATION" "$BACKUP/configuration.nix"

if [[ -f "$SYSTEM_MODULE" ]]; then
    cp -a "$SYSTEM_MODULE" "$BACKUP/hypr-autostart.nix"
fi

if [[ -d "$HYPR_DIR" ]]; then
    cp -a "$HYPR_DIR" "$BACKUP/hyprland"
fi

if [[ -d "$USER_HOME/.config/kitty" ]]; then
    cp -a "$USER_HOME/.config/kitty" "$BACKUP/kitty"
fi

if [[ -d "$USER_HOME/.config/uwsm" ]]; then
    cp -a "$USER_HOME/.config/uwsm" "$BACKUP/uwsm"
fi

if command -v zpool >/dev/null 2>&1 &&
   command -v zfs >/dev/null 2>&1 &&
   zpool list -H rpool >/dev/null 2>&1; then

    SNAPSHOT="rpool@before-mac-binds-$STAMP"

    if zfs snapshot -r "$SNAPSHOT"; then
        echo "Создан ZFS-снимок: $SNAPSHOT"
    else
        echo "Не удалось создать ZFS-снимок, но файловый бэкап создан."
    fi
fi

mkdir -p "$HYPR_LUA_DIR"

cat > "$SYSTEM_MODULE" <<'NIX'
{ pkgs, ... }:

{
  environment.systemPackages = with pkgs; [
    kitty
    fuzzel
    wev
    wl-clipboard
  ];

  programs.firefox.enable = true;

  environment.sessionVariables = {
    MOZ_ENABLE_WAYLAND = "1";
    NIXOS_OZONE_WL = "1";
  };

  programs.hyprland = {
    enable = true;
    withUWSM = true;
    xwayland.enable = true;
  };

  services.greetd = {
    enable = true;

    settings = {
      # Автоматический вход в Hyprland при загрузке.
      initial_session = {
        command = "${pkgs.uwsm}/bin/uwsm start hyprland.desktop";
        user = "w3r";
      };

      # Экран входа после ручного выхода из Hyprland.
      default_session = {
        command = "${pkgs.tuigreet}/bin/tuigreet --time --remember --remember-session";
        user = "greeter";
      };
    };
  };

  # Command работает как Control для обычных сочетаний.
  #
  # Исключения:
  # q     -> Super+Q, перехватывается Hyprland и закрывает окно;
  # space -> Super+Space, открывает launcher;
  # enter -> Super+Enter, открывает Kitty.
  #
  # Физический Control не переназначается, поэтому Control+Space
  # остаётся отдельной комбинацией смены раскладки.
  services.keyd = {
    enable = true;

    keyboards.default = {
      ids = [ "*" ];

      settings = {
        main = {
          leftmeta = "layer(mac)";
          rightmeta = "layer(mac)";
        };

        "mac:C" = {
          q = "M-q";
          space = "M-space";
          enter = "M-enter";
        };
      };
    };
  };

  # Чтобы libinput считал виртуальную клавиатуру keyd внутренней.
  environment.etc."libinput/local-overrides.quirks".text = ''
    [keyd virtual keyboard]
    MatchUdevType=keyboard
    MatchName=keyd virtual keyboard
    AttrKeyboardIntegration=internal
  '';
}
NIX

python3 - "$CONFIGURATION" <<'PY'
from pathlib import Path
import re
import sys

path = Path(sys.argv[1])
text = path.read_text()

import_line = "./hypr-autostart.nix"

if import_line not in text:
    pattern = re.compile(r"(imports\s*=\s*\[\s*\n?)", re.MULTILINE)

    match = pattern.search(text)

    if not match:
        raise SystemExit(
            "Не найден блок imports в configuration.nix. "
            "Добавь ./hypr-autostart.nix вручную."
        )

    insertion = match.group(1) + "    ./hypr-autostart.nix\n"
    text = text[:match.start()] + insertion + text[match.end():]
    path.write_text(text)
PY

if [[ ! -f "$HYPR_MAIN" ]]; then
    cat > "$HYPR_MAIN" <<'LUA'
hl.config({
  general = {
    gaps_in = 5,
    gaps_out = 10,
    border_size = 2,
  },

  decoration = {
    rounding = 10,
  },

  input = {
    kb_layout = "us,ru",

    touchpad = {
      natural_scroll = true,
      tap_to_click = true,
    },
  },

  misc = {
    disable_hyprland_logo = true,
    disable_splash_rendering = true,
  },
})

hl.monitor({
  output = "eDP-1",
  mode = "preferred",
  position = "0x0",
  scale = 2,
})

hl.on("hyprland.start", function()
  hl.exec_cmd(
    "quickshell -p ~/.config/hypr/scripts/quickshell/Shell.qml"
  )
end)
LUA
fi

cat > "$MAC_BINDS" <<'LUA'
-- Mac-подобные сочетания для w3r.
--
-- keyd преобразует обычные Command-комбинации в Ctrl-комбинации.
-- Поэтому Command+C/V/A/X/Z/S/F/T/W работают внутри приложений.
--
-- Command+Q, Command+Space и Command+Enter являются исключениями:
-- keyd передаёт их как Super-комбинации Hyprland.

hl.config({
  input = {
    kb_layout = "us,ru",
    kb_variant = ",",
    kb_options = "",
  },
})

local function safe_unbind(keys)
  pcall(function()
    hl.unbind(keys)
  end)
end

-- Убираем возможные старые конфликтующие сочетания.
safe_unbind("SUPER + Q")
safe_unbind("SUPER + SPACE")
safe_unbind("SUPER + RETURN")
safe_unbind("CTRL + SPACE")

-- Command+Q: закрыть активное окно.
hl.bind(
  "SUPER + Q",
  hl.dsp.window.close(),
  {
    description = "Command+Q: close active application",
  }
)

-- Command+Enter: открыть терминал.
hl.bind(
  "SUPER + RETURN",
  hl.dsp.exec_cmd("kitty"),
  {
    description = "Command+Enter: open Kitty",
  }
)

-- Command+Space: Quickshell launcher, либо Fuzzel как запасной вариант.
hl.bind(
  "SUPER + SPACE",
  hl.dsp.exec_cmd([[
    bash -lc '
      QS="$HOME/.config/hypr/scripts/qs_manager.sh"

      if [ -x "$QS" ]; then
        "$QS" toggle applauncher
      else
        fuzzel
      fi
    '
  ]]),
  {
    description = "Command+Space: application launcher",
  }
)

-- Физический Control+Space: переключение русской/английской раскладки.
hl.bind(
  "CTRL + SPACE",
  hl.dsp.exec_cmd("hyprctl switchxkblayout all next"),
  {
    description = "Control+Space: switch keyboard layout",
  }
)
LUA

python3 - "$HYPR_MAIN" <<'PY'
from pathlib import Path
import re
import sys

path = Path(sys.argv[1])
text = path.read_text()

begin = "-- BEGIN W3R MAC BINDS"
end = "-- END W3R MAC BINDS"

pattern = re.compile(
    re.escape(begin) + r".*?" + re.escape(end) + r"\n?",
    re.DOTALL,
)

text = pattern.sub("", text).rstrip()

block = r'''
-- BEGIN W3R MAC BINDS
package.path =
  "/etc/nixos/config/sessions/hyprland/lua/?.lua;" ..
  package.path

package.loaded["mac-binds"] = nil

local mac_ok, mac_error = pcall(require, "mac-binds")

if not mac_ok then
  print("[mac-binds] " .. tostring(mac_error))
end
-- END W3R MAC BINDS
'''

path.write_text(text + "\n\n" + block.lstrip())
PY

install_user_files() {
    mkdir -p \
        "$USER_HOME/.config/hypr" \
        "$USER_HOME/.config/uwsm" \
        "$USER_HOME/.config/kitty"

    # Убираем старый hyprland.conf, чтобы Hyprland не пытался читать
    # старый формат вместо Lua.
    if [[ -e "$USER_HOME/.config/hypr/hyprland.conf" ||
          -L "$USER_HOME/.config/hypr/hyprland.conf" ]]; then

        mv "$USER_HOME/.config/hypr/hyprland.conf" \
           "$BACKUP/hyprland.conf.user" 2>/dev/null || \
        rm -f "$USER_HOME/.config/hypr/hyprland.conf"
    fi

    rm -f "$USER_HOME/.config/hypr/hyprland.lua"

    ln -s "$HYPR_MAIN" \
        "$USER_HOME/.config/hypr/hyprland.lua"

    touch "$USER_HOME/.config/uwsm/env-hyprland"

    sed -i \
        '/^[[:space:]]*export HYPRLAND_CONFIG=/d' \
        "$USER_HOME/.config/uwsm/env-hyprland"

    echo 'export HYPRLAND_CONFIG="$HOME/.config/hypr/hyprland.lua"' \
        >> "$USER_HOME/.config/uwsm/env-hyprland"

    cat > "$USER_HOME/.config/kitty/mac-bindings.conf" <<'KITTY'
# Command+C приходит от keyd как Ctrl+C.
# При выделенном тексте копирует; без выделения отправляет SIGINT.
map --allow-fallback=shifted,ascii ctrl+c copy_or_interrupt

# Command+V приходит от keyd как Ctrl+V.
map --allow-fallback=shifted,ascii ctrl+v paste_from_clipboard

# Старые Linux-сочетания тоже оставляем.
map --allow-fallback=shifted,ascii ctrl+shift+c copy_to_clipboard
map --allow-fallback=shifted,ascii ctrl+shift+v paste_from_clipboard
KITTY

    KITTY_CONFIG="$USER_HOME/.config/kitty/kitty.conf"

    # Если Home Manager сделал kitty.conf ссылкой на Nix Store,
    # превращаем его в обычный редактируемый файл с тем же содержимым.
    if [[ -L "$KITTY_CONFIG" ]]; then
        cp -L "$KITTY_CONFIG" "$BACKUP/kitty.conf.from-home-manager"
        rm -f "$KITTY_CONFIG"
        cp "$BACKUP/kitty.conf.from-home-manager" "$KITTY_CONFIG"
    fi

    touch "$KITTY_CONFIG"

    if ! grep -qF "include mac-bindings.conf" "$KITTY_CONFIG"; then
        printf '\n# w3r macOS-like shortcuts\ninclude mac-bindings.conf\n' \
            >> "$KITTY_CONFIG"
    fi

    chown -R "$TARGET_USER:$USER_GROUP" \
        "$USER_HOME/.config/hypr" \
        "$USER_HOME/.config/uwsm" \
        "$USER_HOME/.config/kitty"
}

install_user_files

echo
echo "Проверяю Nix-конфигурацию..."

cd "$NIXOS_DIR"

nix flake check --no-build

echo
echo "Выполняю безопасную тестовую сборку..."

nixos-rebuild dry-build --flake "$NIXOS_DIR#$FLAKE_HOST"

echo
echo "Устанавливаю новую загрузочную генерацию..."

nixos-rebuild boot --flake "$NIXOS_DIR#$FLAKE_HOST"

# На случай, если Home Manager во время сборки восстановил свои ссылки.
install_user_files

echo
echo "Готово."
echo
echo "Бэкап:"
echo "  $BACKUP"
echo
echo "После перезагрузки:"
echo "  Command+C/V     — копировать/вставить"
echo "  Command+Q       — закрыть активное приложение"
echo "  Command+Enter   — Kitty"
echo "  Command+Space   — меню приложений"
echo "  Control+Space   — переключение RU/EN"
echo
echo "Для отката выбери предыдущую генерацию в systemd-boot."
echo

read -r -p "Перезагрузить MacBook сейчас? [y/N] " answer

case "$answer" in
    y|Y|yes|YES|да|Да|ДА)
        reboot
        ;;
    *)
        echo "Перезагрузи позже командой: sudo reboot"
        ;;
esac
