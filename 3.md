# 1. Создаём реальный Lua-конфиг по адресу, на который уже ведёт ссылка
sudo install -d -m 0755 /etc/nixos/config/sessions/hyprland

sudo tee /etc/nixos/config/sessions/hyprland/hyprland.lua >/dev/null <<'EOF'
-- Рабочий базовый конфиг Hyprland 0.55+

hl.config({
  general = {
    gaps_in = 5,
    gaps_out = 10,
    border_size = 2,
  },

  decoration = {
    rounding = 10,
  },

  input = {
    kb_layout = "us,ru",
    kb_options = "grp:win_space_toggle",

    touchpad = {
      natural_scroll = true,
      tap_to_click = true,
    },
  },

  misc = {
    disable_hyprland_logo = true,
    disable_splash_rendering = true,
  },
})

hl.monitor({
  output = "eDP-1",
  mode = "preferred",
  position = "0x0",
  scale = 2,
})

hl.bind("SUPER + RETURN", hl.dsp.exec_cmd("kitty"))
hl.bind("SUPER + F", hl.dsp.exec_cmd("firefox"))
hl.bind("SUPER + Q", hl.dsp.window.close())
hl.bind("SUPER + M", hl.dsp.exec_cmd("uwsm stop"))

hl.on("hyprland.start", function()
  hl.exec_cmd("quickshell -p $HOME/.config/hypr/scripts/quickshell/Shell.qml")
end)
EOF

# 2. Удаляем старый ошибочный конфиг
mkdir -p ~/.config/hypr/backup

if [ -e ~/.config/hypr/hyprland.conf ]; then
  mv ~/.config/hypr/hyprland.conf \
    ~/.config/hypr/backup/hyprland.conf.$(date +%Y%m%d-%H%M%S)
fi

# 3. Пересоздаём ссылку на существующий файл
rm -f ~/.config/hypr/hyprland.lua

ln -s /etc/nixos/config/sessions/hyprland/hyprland.lua \
  ~/.config/hypr/hyprland.lua

# 4. Исправляем переменную UWSM
mkdir -p ~/.config/uwsm
touch ~/.config/uwsm/env-hyprland

sed -i '/^[[:space:]]*export HYPRLAND_CONFIG=/d' \
  ~/.config/uwsm/env-hyprland

echo 'export HYPRLAND_CONFIG="$HOME/.config/hypr/hyprland.lua"' \
  >> ~/.config/uwsm/env-hyprland

# 5. Проверяем, что файл теперь реально читается
echo "Путь:"
readlink -f ~/.config/hypr/hyprland.lua

echo "Начало конфига:"
head -n 8 ~/.config/hypr/hyprland.lua
